  let i = 0;
  return {
    next() {
      if (i < val._keys.length) {

  };
};

const printImmutableSeq = (val, config, indentation, depth, refs, printer) => {
  const name = getImmutableName('Seq');

    return printAsLeaf(name);

      name +

            refs,
